  \section*{Introduction}
  The program \ty{land} labels the internal nodes of a Newick tree. This
  makes it easy to subsequently pick individual clades in the tree with
  \ty{pickle}. The program can also remove labels from the internal
  nodes of a labeled tree and/or from the leaves.

#+begin_src go <<Functions, Pr. \ref{pr:lan}>>=
  func labelTree(v *nwk.Node, c int, pr, su string,
	  ri, rl bool) int {
	  if v == nil {
		  return c
	  }
	  //<<Label node, Pr. \ref{pr:lan}>>
	  c = labelTree(v.Child, c, pr, su, ri, rl)
	  c = labelTree(v.Sib, c, pr, su, ri, rl)
	  return c
  }
