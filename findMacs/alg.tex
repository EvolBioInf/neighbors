\REQUIRE{$T=\{t_1,t_2,...,t_k\}$} \COMMENT{Target sequences.}
\REQUIRE{$N=\{n_1,n_2,...,n_l\}$} \COMMENT{Neighbor sequences.}
\ENSURE{$M$} \COMMENT{Markers, regions ubiquitous in $T$ and
  absent from $N$.}
\STATE{$M\leftarrow t_1$}
\FOR{$i\leftarrow 1$ to $l$}
\STATE{$M\leftarrow M\setminus (t_1\cap n_i)$}
\ENDFOR
\FOR{$i\leftarrow 2$ to $k$}
\STATE{$M\leftarrow M\setminus(t_1\setminus(t_1\cap t_i))$}
\ENDFOR
